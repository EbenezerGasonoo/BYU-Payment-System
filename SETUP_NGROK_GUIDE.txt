📥 STEP 1: DOWNLOAD NGROK (1 minute)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  3. Extract the ZIP file
  
  4. Move ngrok.exe to:
     I:\Projects\BYU Payment System\


🔑 STEP 2: GET FREE ACCOUNT (30 seconds)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  1. Go to: https://dashboard.ngrok.com/signup
  
  2. Sign up with:
     → Google (instant)
     → GitHub (instant)
     → Email (quick)
  
  3. After signup, you'll see your dashboard


🎫 STEP 3: GET YOUR TOKEN (30 seconds)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  1. Go to: https://dashboard.ngrok.com/get-started/your-authtoken
  
  2. Copy the token (looks like):
     2abc123def456ghi789jkl012mno345pqr678stu
  
  3. Keep it handy for next step


⚙️ STEP 4: SETUP (One-time, 30 seconds)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  Open PowerShell in your project folder:
  
  cd "I:\Projects\BYU Payment System"
  .\ngrok config add-authtoken YOUR_TOKEN_HERE
  
  (Replace YOUR_TOKEN_HERE with the token you copied)


🚀 STEP 5: EXPOSE YOUR APP! (Instant)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
